Value dev-network

Network (VPC)

dev-network:
  new(name: "dev-network-sam"
      region: Us-west-2
      cidr: "10.0.0.0/16"
      publicSubnets:
        [(A, "10.0.1.0/24")
        ,(B, "10.0.2.0/24")]
      privateSubnets: [])

Value dev-sam-sg

Web-facing SecurityGroup

dev-sam-sg:
  SecurityGroup(description:
                  "Allow http/s traffic from the Internet"
                ipPermissions: [webHTTPS
                               ,webHTTP
                               ,ssh]
                ipPermissionsEgress: None
                tags: [web-app-tag]
                vpc: dev-network.vpc)

Value webHTTP

webHTTP:
  IpPermission(ipProtocol: "tcp"
               fromPort: 80
               toPort: 80
               prefixLists: None
               target: internetCIDR)

Value webHTTPS

webHTTPS:
  IpPermission(ipProtocol: "tcp"
               fromPort: 443
               toPort: 443
               prefixLists: None
               target: internetCIDR)

Value ssh

ssh:
  IpPermission(ipProtocol: "tcp"
               fromPort: 22
               toPort: 22
               prefixLists: None
               target: internetCIDR)

Value internetCIDR

Convenience

internetCIDR:
  IpRanges([IpRange(cidrIp:
                      "0.0.0.0/0")])

Value web-app-tag

Tags

web-app-tag:
  Tag(key: "Environment"
      value: "dev-sam")
